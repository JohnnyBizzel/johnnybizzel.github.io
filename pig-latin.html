Pig Latin converter.

Translate the provided string to pig latin. 

function translatePigLatin(str) {
  var returnArr = [];
  strArr = str.split('');

  	// test regex for Vowel:
  	var patt = /A|a|E|e|I|i|O|o|U|u/;
  	var foundVowel = false;
  	var c = 0;
  	var consGrp = [];
  			
  	while(!foundVowel) {
  				
  		if (patt.test(strArr[0])) {
		
		// found vowel is the first char	
			foundVowel = true;			
  	  	// after = after.replaceAt(0, after.charAt(0).toUpperCase());
  		  	for (i = 0; i < strArr.length; i ++) {
  				returnArr.push(strArr[i]);
  			}
  			returnArr.push('w');
  			returnArr.push('a');
  			returnArr.push('y');
  		} else {
  			// consonant
  				
  			if (patt.test(strArr[c])) {
  				// check for next vowel
  				foundVowel = true;
  				for (i = c; i < strArr.length; i ++) {
  					returnArr.push(strArr[i]);
  				}
  				returnArr.push(consGrp.join(''));
  				returnArr.push('a');
  				returnArr.push('y');
  				} else {
  					consGrp.push(strArr[c]);
  				}
  				
  		}
  	c++;
  	}
  	
  return returnArr.join("");;
}
