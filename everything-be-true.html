Everything Be True

Check if the predicate (second argument) is truthy on all elements of a collection (first argument). 
truthCheck([{"user": "Tinky-Winky", "sex": "male"}, {"user": "Dipsy", "sex": "male"}, {"user": "Laa-Laa", "sex": "female"}, {"user": "Po", "sex": "female"}],
 "sex");

truthCheck([{"user": "Tinky-Winky", "sex": "male"}, {"user": "Dipsy"}, {"user": "Laa-Laa", "sex": "female"}, {"user": "Po", "sex": "female"}],
 "sex"); 

truthCheck([{"single": "yes"}], "single");

truthCheck([{"single": ""}, {"single": "double"}], "single");

truthCheck([{"single": "double"}, {"single": NaN}], "single"); 

function truthCheck(collection, pre) {
  // Is everyone being true?
  var isTrue = true;  // assume all have the prop u checking
  collection.forEach(function(val) {
    if (!val.hasOwnProperty(pre))  { 
    	
      isTrue = false;   // if at least one of the object doesn't have the property you are checking... 
    } 
    // Test for illegal values
    if (!val[pre]) {
    	isTrue = false; 
    }
    	 
  });
	return isTrue;
}
