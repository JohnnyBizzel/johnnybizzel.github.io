Binary Agent

Return an English translated sentence of the passed binary string. 

function binaryAgent(str) {
  var items = str.split(' ');
  var finalString = "";
  var strLen = items.length;
  var it;
  for (it=0; it < strLen; it++) {
  		 var s = items[it].split('');    // my binary "number" as a string
		 var dec = 0;
		 var sLen = s.length;
		 var i;
		 for( i=0; i < sLen; i++ ) {
			  // we start with the least significant digit, and work our way to the left
			  if( s[s.length-i-1] == '0' ) continue;
			    dec += Math.pow( 2, i );
			    
		 }
    	finalString += String.fromCharCode(dec);	 
		  	
  }
		  
  return finalString;
}
