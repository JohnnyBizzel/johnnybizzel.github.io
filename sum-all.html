Sum all numbers in a range.

function getSum(total, num) {
    return total + num;
}

function sumAll(arr) {
 	
 	arr.sort(function(a, b){return a-b});
	var  minItem = Math.max(arr[0]);
 	var maxItem = Math.min(arr[1]); 	
	var completeArr =[]; 	
 	for (i=minItem; i <= maxItem; i++) {
		 	completeArr.push(i);
 	} 
  return completeArr.reduce(getSum);
}
