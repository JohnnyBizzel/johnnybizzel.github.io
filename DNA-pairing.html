function pairElement(str) {
  var strArr = str.split('');
  var retArr = [];
  for (var i = 0; i < strArr.length; i ++) {
	   switch(strArr[i]) {
	    case 'A':
	        // A -> A,T
	        var a_t = ["A","T"];
	        retArr.push(a_t);
	        break;
	    case 'T':
	        // T -> T,A
	        var ta = ["T","A"];
	        retArr.push(ta);

	        break;
	    case 'G':
	        // G -> G,C
	        var gc = ["G","C"];
	        retArr.push(gc);

	        break;
	    case 'C':
	        // C -> C,G
	        var cg = ["C","G"];
	        retArr.push(cg);

	        break;

	    default:
	        // default
		}
  }


  // Only change code above this line
  return JSON.stringify(retArr);
}
