function telephoneCheck(str) {
  if (str.charAt(0) == '(' && str.charAt(4) != ')') return false;
 
  if (str.replace(/[^0-9]/g,"").length === 10) {
        var re = /([0-9]{3}-?|\([\d]{3}\))[0-9]{3}-?[0-9]{4}/g;

        var valid = reggie.test(str);
