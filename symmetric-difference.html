Symmetric Difference

Create a function that takes two or more arrays and returns an array of the symmetric difference of the provided arrays. 

function combineAry(previousArr, currentArr) {
	// Return the difference of the arrays.
	var tempAry = [];
	var hs = {}; // initialise object
	// hash[1] += 1
	// hash  ==> Object {3: 5, 2: 1}
	previousArr = previousArr.filter(function(elem, pos) {
	  return previousArr.indexOf(elem) == pos;
	});
	currentArr = currentArr.filter(function(elem, pos) {
	  return currentArr.indexOf(elem) == pos;
	});

	for (var i = 0; i < previousArr.length; i++) {
		var idx = previousArr[i];
		if( hs[parseInt(idx)] > 0) {
	   	hs[parseInt(idx)] += 1;
	   } else {
	   	hs[parseInt(idx)] = 1;
	   }
	}
	for (var j = 0; j < currentArr.length; j++) {
	   if( hs[parseInt(currentArr[j])] > 0) {
	   	hs[parseInt(currentArr[j])] += 1;
	   } else {
	   	hs[parseInt(currentArr[j])] = 1;
	   }
	
	}

	for (var key in hs) {
   	if (hs[parseInt(key)] === 1) {
		   tempAry.push(parseInt(key));
   	}

	}

	return tempAry;
	
}


function sym(args) {
	var testAry = [];
	testAry = Array.from(arguments).reduce(combineAry);

		
  return (testAry);
