Missing letters.

Find the missing letter in the passed letter range and return it. eg) abce should return "d" 

function fearNotLetter(str) {
	var returnLetter;
	var code;
	var prevCode;
	prevCode = str.charCodeAt(0);
			
	for (i = 1; i < str.length; i ++) {
		code = str.charCodeAt(i);
		if (code == prevCode + 1)
		{	// in sequence
			prevCode = code;
		} else {
			returnLetter = String.fromCharCode(code - 1);
			prevCode = code;
		}
   }

  return returnLetter;
}
