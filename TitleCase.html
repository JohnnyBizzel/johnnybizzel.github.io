Title Case a Sentence.

function titleCase(str) {
  var strArr = str.split(' ');
  if (strArr.length > 1) {
     for (i = 0; i < strArr.length; i++) {
         var s = strArr[i].toLowerCase();
         var schars = s.split('');
         for (j = 0; j < schars.length; j++) {
             schars[j] = schars[j].toLowerCase();
         }
         schars[0] = schars[0].toUpperCase();
         s= schars.join('');
         strArr[i] = s;
     }
    
  }
    
      return strArr.join(' ');
}
