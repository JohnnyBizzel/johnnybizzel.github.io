Sum Prime numbers

function check_prime(a)
{
   var c = 2;
   for ( c = 2 ; c <= a - 1 ; c++ )
   { 
      if ( a % c === 0 )
	   	return false;
   }
   if ( c === a )
      return true;
}

function makePrimeArray(numMax){
	 var x = 0; // running total
    for(var primeArray = [], k=2; k<=numMax;k++ ){
    		if (check_prime(k)) primeArray.push(k);        
    }
    return primeArray;
}

function sumPrimes(num) {

	var primesAry = makePrimeArray(num);
	var runningTot = 0;
	for (var i = 0; i < primesAry.length; i++) {
		if (primesAry[i] <= num) {
			runningTot += primesAry[i];
				
		} else {
		// stop
			num = runningTot;
			return num;
		}
	}
	
   return runningTot;
}
