Array into chunks.

function chunkArrayInGroups(arr, size) {
  // Break it up.
  var slices = Math.floor(arr.length / size) + 1;
  var lastChunk = arr.length % size;	
    
  var slicedArr = [];
  var pos = 0;
  for (i = 0; i < slices; i ++) {
	var chunk = [];
  	if ((size + pos) > arr.length )
  	{
		// use last chunk
		if (lastChunk > 0) {
			chunk = (arr.slice(pos, pos + size));
			slicedArr.push(chunk);		
		}
				  		
  	} else {
  		chunk = (arr.slice(pos, pos + size));	
  		slicedArr.push(chunk);		
  	}
		
	pos += size;  
  }
  
  return (slicedArr);
}
