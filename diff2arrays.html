Find items not in both arrays 

function diffArray(arr1,arr2) {
   var finalArr = [];
   var longArray = [];
   var shortArray = [];
   if (arr1.length > arr2.length) {
	  longArray = arr1;
	  shortArray = arr2;
   } else {
	  longArray = arr2;
	  shortArray = arr1;	
   }
   
   if (JSON.stringify(arr1)==JSON.stringify(arr2)) {
  		return finalArr;
   }

	 var found = false;
	 var i;
	 var lngAryLen = longArray.length;	
	 var shoAryLen = shortArray.length
    for (i = 0; i < lngAryLen; i++) {
        for (j = 0; j < shoAryLen.length; j++) {
        		if (longArray[i] === shortArray[j]) {
        			found = true;
        			continue;
        		}
        		
        }
        if (!found) finalArr.push(longArray[i]);        		
       found = false;
    }
    found = false;
    for (i = 0; i < shoAryLen; i++) {
        for (j = 0; j < lngAryLen; j++) {
        		if (longArray[j] === shortArray[i]) {
        			found = true;
        			continue;
        		}
        		
        }
        if (!found) finalArr.push(shortArray[i]);        		
       found = false;
    }
    return finalArr;
}
