function makeFibArray(numMax){
	var x = 0; // running total
    for(var fibArray = [1], i=0,j=1,k=0; x < numMax; i=j,j=x,k++ ){
        x=i+j;
        fibArray.push(x);
    }
    return fibArray;
}

function sumFibs(num) {

	var fibArray = makeFibArray(num);
	var runningTot = 0;
	for (var i = 0; i < fibArray.length; i++) {
		if (fibArray[i] <= num) {
			if (fibArray[i] % 2 === 1) {
				runningTot += fibArray[i];
			}			
		} else {
		// stop
			num = runningTot;
			return num;
		}
	}

	
   return runningTot;
}
