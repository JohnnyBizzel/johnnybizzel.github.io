Pairwise

Given an array arr, find element pairs whose sum equal the second argument arg and return the sum of their indices.
If multiple pairs are possible that have the same numeric elements but different indices, return the smallest sum of indices. Once an element has been used, it cannot be reused to pair with another.

function pairwise(arr, arg) {
	var idxAry = [];
	var sum = 0;
	if (arr.length < 1) { return 0; } 
	else {
			// make an array of indexes as we need these values later.
		for(var a = 0; a < arr.length; a ++) {
			idxAry.push(a);		
		}
			
		// looping from first element
		for(i = 0; i < arr.length; i++) {
		  	//Looping from second element by setting first element  constant
		  	for(j = i + 1; j < arr.length; j++) {
			// Check whether the sum is equal to arg
			  if(arr[i] + arr[j] == arg) {
			   	//Add the indices
			     if (!isNaN(idxAry[i]) && !isNaN(idxAry[j])) {
				   sum += idxAry[i] + idxAry[j];
				     //Set the indices to NaN so that they can't be used in next iteration
				     idxAry[i] = NaN;
				     idxAry[j] = NaN;		     
			     }
			   }
			}
		}
	}	
  	return sum;
}
