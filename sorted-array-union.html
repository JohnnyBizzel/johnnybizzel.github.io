function uniteUnique(arr) {
	var finalAry = [];
	
	finalAry = Array.from(arguments).reduce(function(previousArr, currentArr) {
		return previousArr.concat(currentArr);
	});
		
		// filter out duplicates
	var combined = [finalAry[0]];
	var found = false;
	var indexOutput = 1;
		
	// Found solution on http://stackoverflow.com/questions/3350641/array-remove-duplicate-elements by dhayyati
	for (i = 1; i < finalAry.length; i ++) {
			var j = 0; 
			while ((combined[j] !== finalAry[i]) && j < indexOutput) {
					j++;
			}
			if(j === indexOutput) {
				combined[indexOutput] = finalAry[i];
				indexOutput ++; 												
			}												
	}
				
	  return combined.join(" ");								
}
