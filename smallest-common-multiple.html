function createArray(arr) {

    var a = arr[0];
    var b = arr[1];
    var newArr = [];

    if (a > b) {
        
        newArr.push(b);
        while (a > b) {
            b++;
            newArr.push(b);         
        }
    } else {
        newArr.push(a);
        while (b > a) {
            a++;
            newArr.push(a);         
        }
    } 
    return newArr;
       
}

function lcm(t1,t2)
{   var f = hcf(t1,t2);
    return t1 *t2 /f;
}

// Highest common factor
function hcf(a,b) {
        var gcd=1;
        if (a > b) {
            a = a + b;
            b = a - b;
            a = a - b;
        }
        if ((b ==(Math.round(b / a)) * a))
            { gcd = a; }
        else { 
            for (var i=Math.round(a / 2); i > 1; i-- )
            {  if((a == (Math.round(a / i)) * i))
                    if((b == (Math.round(b / i)) * i))
                    {   gcd=i;  
                        i=-1;
                    }
            }
        }
        return gcd;
}
