Where art thou.

Looking up objects in a collection of objects. 


function whereAreYou(collection, source) {
  // What's in a name?
  var sourceKeys = Object.keys(source);
  var arr = [];

  // Only change code below this line
  //Iterate through array of objects and check for the property of source object
  arr = collection.filter(function(obj){
    var foundItems = sourceKeys.length;
    for (var j = 0; j < sourceKeys.length; j++){
      
      if (obj.hasOwnProperty(sourceKeys[j]) && obj[sourceKeys[j]] === source[sourceKeys[j]]) {
         foundItems--;
      }

    }
    if (foundItems === 0) { return true; } else { return false ;} 
    
  });
 
  return (arr);
}
