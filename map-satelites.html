Return a new array that transforms the element's average altitude into their orbital periods. 

function orbitalPeriod(arr) {
  var GM = 398600.4418;
  var earthRadius = 6367.4447;
  var twoPie = 2 * Math.PI;
   var retArr = [];
   for (var i = 0; i < arr.length; i++) {
	  var solution = twoPie * Math.sqrt(Math.pow((earthRadius + arr[i].avgAlt), 3) / GM);
	
	  // make a new return object	 
	  var retObj = {name: arr[i].name, orbitalPeriod: Math.round(solution) };
	  retArr.push(retObj);
   }
  
  return (retArr);
}
